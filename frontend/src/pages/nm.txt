    const targetPlayerCount = Math.pow(2, totalRounds);
    const matchesInFirstRound = targetPlayerCount / 2;
    const byesNeeded = targetPlayerCount - playerCount;
    const gamesByRound = Array(totalRounds).fill().map(() => []);
    const byePositions = new Set();
    if (byesNeeded > 0) {
      for (let i = 0; i < byesNeeded; i++) {
        const pos = matchesInFirstRound - 1 - Math.floor(i * matchesInFirstRound / byesNeeded);
        byePositions.add(pos);
      }
    }
    let playerIndex = 0;
    const firstRoundGames = [];
    const directAdvancers = [];
    for (let slot = 0; slot < matchesInFirstRound; slot++) {
      if (byePositions.has(slot)) {
        if (playerIndex < playerCount) {
          const byePlayer = p(playerIndex);
          playerIndex++;
          byePlayer.slot = slot;
          directAdvancers.push(byePlayer);
        }
      } else {

            pool: poolLabel?.replace('Pool ', '') || '',

      }

      for (let i = 0; i < currentAdvancers.length; i += 2) {
        const homeAdvancer = currentAdvancers[i];
        const awayAdvancer = i + 1 < currentAdvancers.length ? currentAdvancers[i + 1] : null;
        const game = {
          id: matchId++,
          name: getRoundName(roundNumber, totalRounds, poolLabel),
          round: roundNumber,
          scheduled: Date.now(),
          pool: poolLabel?.replace('Pool ', '') || '',
          sides: { home: null, away: null }
        };
        game.sides.home = homeAdvancer ? (homeAdvancer.round ? src(homeAdvancer) : homeAdvancer) : null;
        if (awayAdvancer) game.sides.away = awayAdvancer.round ? src(awayAdvancer) : awayAdvancer;
        nextRound.push(game);
      }
      gamesByRound[roundNumber - 1] = nextRound;
      currentAdvancers = nextRound;
      roundNumber++;
    }
    return { finalGame: currentAdvancers[0] || null, gamesByRound };
  };

  // === GENERATE ALL BRACKETS ===
  useEffect(() => {
    if (players.length === 0) return;

    console.log('Bracket generation triggered due to players change');

    // Preserve locked brackets fully
    const preservedLocked = brackets.filter(b => lockedBrackets.has(b.key));
    const preservedOutcomes = {};
    preservedLocked.forEach(b => {
      if (bracketsOutcomes[b.key]) {
        preservedOutcomes[b.key] = { ...bracketsOutcomes[b.key] };
      }
    });

    const smartSeedPlayers = (playersList) => {
      if (playersList.length <= 2) return [...playersList];

      const teamGroups = {};
      playersList.forEach(p => {
        const team = p.team?.trim() || 'No Team';
        if (!teamGroups[team]) teamGroups[team] = [];
        teamGroups[team].push(p);
      });

      const sortedTeams = Object.entries(teamGroups)
        .sort((a, b) => b[1].length - a[1].length);

      const totalSlots = Math.pow(2, Math.ceil(Math.log2(playersList.length)));
      const seeded = new Array(totalSlots);

      let topIndex = 0;
      let bottomIndex = totalSlots - 1;

      sortedTeams.forEach(([team, teamPlayers]) => {
        teamPlayers.forEach((player, i) => {
          if (i % 2 === 0) {
            while (topIndex < totalSlots && seeded[topIndex]) topIndex++;
            if (topIndex < totalSlots / 2) {
              seeded[topIndex] = player;
              topIndex += 2;
              if (topIndex >= totalSlots / 2) topIndex = 1;
            }
          } else {
            while (bottomIndex >= totalSlots / 2 && seeded[bottomIndex]) bottomIndex--;
            if (bottomIndex >= totalSlots / 2) {
              seeded[bottomIndex] = player;
              bottomIndex -= 2;
              if (bottomIndex < totalSlots / 2) bottomIndex = totalSlots - 2;
            }
          }
        });
      });

      let remaining = playersList.filter(p => !seeded.includes(p));
      let fillIndex = 0;
      for (let i = 0; i < totalSlots; i++) {
        if (!seeded[i] && remaining[fillIndex]) {
          seeded[i] = remaining[fillIndex++];
        }
      }

      return seeded.filter(Boolean);
    };

    try {
      const grouped = players.reduce((acc, p) => {
        const key = `${p.gender}_${p.ageCategory}_${p.weightCategory}`;
        if (!acc[key]) acc[key] = [];
        acc[key].push(p);
        return acc;
      }, {});

      const generatedBrackets = [];

      Object.entries(grouped).forEach(([key, groupPlayers]) => {
        const [gender, ageCategory, weightCategory] = key.split('_');
        const categoryPlayerCount = groupPlayers.length;

        if (categoryPlayerCount <= 16) {
          const seededPlayers = smartSeedPlayers([...groupPlayers]);
          const { finalGame, gamesByRound } = generateSingleEliminationGameStructure(seededPlayers);

          generatedBrackets.push({
            key,
            sanitizedKey: sanitizeId(key),
            gender,
            ageCategory,
            weightCategory,
            playerCount: categoryPlayerCount,
            categoryPlayerCount,
            shuffledPlayers: seededPlayers,
            game: finalGame,
            gamesByRound,
            outcomes: {}
          });
        } else {
          const minPools = Math.ceil(categoryPlayerCount / 16);
          let numPools = 1;
          while (numPools < minPools) numPools *= 2;

          const poolSizes = Array(numPools).fill(Math.floor(categoryPlayerCount / numPools));
          const extras = categoryPlayerCount % numPools;
          for (let i = 0; i < extras; i++) poolSizes[i]++;

          const allSeededPlayers = smartSeedPlayers([...groupPlayers]);
          let playerStart = 0;
          const pools = [];

          for (let i = 0; i < numPools; i++) {
            const poolLabelChar = String.fromCharCode(65 + i);
            const poolPlayersRaw = allSeededPlayers.slice(playerStart, playerStart + poolSizes[i]);
            const poolPlayers = poolPlayersRaw.length > 2 ? smartSeedPlayers(poolPlayersRaw) : poolPlayersRaw;
            playerStart += poolSizes[i];

            const poolStruct = generateSingleEliminationGameStructure(poolPlayers, `Pool ${poolLabelChar}`);

            pools.push({
              key: `${key}_Pool${poolLabelChar}`,
              sanitizedKey: sanitizeId(`${key}_Pool${poolLabelChar}`),
              gender,
              ageCategory,
              weightCategory,
              playerCount: poolPlayers.length,
              categoryPlayerCount,
              shuffledPlayers: poolPlayers,
              game: poolStruct.finalGame,
              gamesByRound: poolStruct.gamesByRound,
              pool: poolLabelChar,
              outcomes: {}
            });
          }

          const dummyPlayoffPlayers = pools.map((_, idx) => ({
            name: `Winner Pool ${String.fromCharCode(65 + idx)}`,
            team: ''
          }));

          let playoffStruct = generateSingleEliminationGameStructure(dummyPlayoffPlayers);
          let poolIndex = 0;
          playoffStruct.gamesByRound[0].forEach(game => {
            game.sides.home = { sourceGame: pools[poolIndex].game, score: { score: null }, pool: pools[poolIndex].pool };
            poolIndex++;
            if (game.sides.away) {
              game.sides.away = { sourceGame: pools[poolIndex].game, score: { score: null }, pool: pools[poolIndex].pool };
              poolIndex++;
            }
          });

          generatedBrackets.push(...pools);
          generatedBrackets.push({
            key: `${key}_PoolFinal`,
            sanitizedKey: sanitizeId(`${key}_PoolFinal`),
            gender,
            ageCategory,
            weightCategory,
            playerCount: numPools,
            categoryPlayerCount,
            shuffledPlayers: [],
            game: playoffStruct.finalGame,
            gamesByRound: playoffStruct.gamesByRound,
            pool: 'Final',
            outcomes: {}
          });
        }
      });

      generatedBrackets.sort((a, b) => {
        const g = genderOrder.indexOf(a.gender) - genderOrder.indexOf(b.gender);
        if (g !== 0) return g;
        const aIdx = ageCategoryOrder.indexOf(a.ageCategory);
        const bIdx = ageCategoryOrder.indexOf(b.ageCategory);
        if (aIdx !== bIdx) return aIdx - bIdx;
        const weightCmp = a.weightCategory.localeCompare(b.weightCategory);
        if (weightCmp !== 0) return weightCmp;
        const poolOrder = a.pool ? (a.pool === 'Final' ? 999 : a.pool.charCodeAt(0)) : 0;
        const pbOrder = b.pool ? (b.pool === 'Final' ? 999 : b.pool.charCodeAt(0)) : 0;
        return poolOrder - pbOrder;
      });

      const finalBrackets = [...preservedLocked];

      generatedBrackets.forEach(newB => {
        const lockedIndex = finalBrackets.findIndex(b => b.key === newB.key && lockedBrackets.has(b.key));
        if (lockedIndex !== -1) return;

        const existingIndex = finalBrackets.findIndex(b => b.key === newB.key);
        if (existingIndex !== -1) {
          finalBrackets[existingIndex] = newB;
        } else {
          finalBrackets.push(newB);
        }
      });

      finalBrackets.sort((a, b) => {
        const g = genderOrder.indexOf(a.gender) - genderOrder.indexOf(b.gender);
        if (g !== 0) return g;
        const aIdx = ageCategoryOrder.indexOf(a.ageCategory);
        const bIdx = ageCategoryOrder.indexOf(b.ageCategory);
        if (aIdx !== bIdx) return aIdx - bIdx;
        const weightCmp = a.weightCategory.localeCompare(b.weightCategory);
        if (weightCmp !== 0) return weightCmp;
        const poolOrder = a.pool ? (a.pool === 'Final' ? 999 : a.pool.charCodeAt(0)) : 0;
        const pbOrder = b.pool ? (b.pool === 'Final' ? 999 : b.pool.charCodeAt(0)) : 0;
        return poolOrder - pbOrder;
      });

      const bracketsChanged = JSON.stringify(finalBrackets.map(b => b.key)) !== JSON.stringify(brackets.map(b => b.key));
      if (bracketsChanged) {
        setBrackets(finalBrackets);
      }

      const newOutcomes = { ...preservedOutcomes };
      finalBrackets.forEach(br => {
        if (!lockedBrackets.has(br.key)) {
          newOutcomes[br.key] = {};
        }
      });

      const outcomesChanged = JSON.stringify(newOutcomes) !== JSON.stringify(bracketsOutcomes);
      if (outcomesChanged) {
        setBracketsOutcomes(newOutcomes);
      }

    } catch (err) {
      console.error('Bracket generation failed:', err);
      setError('Failed to generate brackets.');
    }

const shuffleBracket = (bracketKey) => {
  if (lockedBrackets.has(bracketKey)) {
    console.log('Shuffle blocked: bracket is locked');
    return; // Locked bracket को shuffle नहीं करना
  }

  setBrackets(prevBrackets => {
    if (!prevBrackets || prevBrackets.length === 0) return prevBrackets;

    const bracketIndex = prevBrackets.findIndex(b => b.key === bracketKey);
    if (bracketIndex === -1) return prevBrackets;

    const bracket = prevBrackets[bracketIndex];

    // Pool Final को shuffle नहीं करना
    if (bracket.pool === 'Final') return prevBrackets;

    let newBrackets = [...prevBrackets];

    if (bracket.pool) {
      // === POOL CATEGORY SHUFFLE ===
      const baseKey = bracket.key.replace(/_Pool.*$/, '');
      const poolBrackets = newBrackets
        .filter(b => b.key.startsWith(`${baseKey}_Pool`) && b.pool !== 'Final')
        .sort((a, b) => a.pool.charCodeAt(0) - b.pool.charCodeAt(0));

      if (poolBrackets.length === 0) return prevBrackets;

      // सभी pool players को collect करो
      let allPlayers = [];
      const poolSizes = [];
      poolBrackets.forEach(pb => {
        allPlayers.push(...pb.shuffledPlayers);
        poolSizes.push(pb.shuffledPlayers.length);
      });

      // Random shuffle
      allPlayers.sort(() => Math.random() - 0.5);

      // Players को वापस pools में distribute करो
      let start = 0;
      const newPoolBrackets = poolBrackets.map((pb, i) => {
        const size = poolSizes[i];
        const newPlayers = allPlayers.slice(start, start + size);
        start += size;

        const { finalGame, gamesByRound } = generateSingleEliminationGameStructure(newPlayers, `Pool ${pb.pool}`);

        return {
          ...pb,
          shuffledPlayers: newPlayers,
          game: finalGame,
          gamesByRound,
          outcomes: {} // Shuffle पर outcomes clear
        };
      });

      // Update pool brackets in newBrackets
      newPoolBrackets.forEach(npb => {
        const idx = newBrackets.findIndex(b => b.key === npb.key);
        if (idx !== -1) {
          newBrackets[idx] = npb;
        }
      });

      // === Regenerate Pool Final bracket ===
      const poolFinalIndex = newBrackets.findIndex(b => b.key === `${baseKey}_PoolFinal`);
      if (poolFinalIndex !== -1) {
        const dummyPlayoffPlayers = newPoolBrackets.map((_, idx) => ({
          name: `Winner Pool ${String.fromCharCode(65 + idx)}`,
          team: ''
        }));

        let playoffStruct = generateSingleEliminationGameStructure(dummyPlayoffPlayers);

        let poolIndex = 0;
        playoffStruct.gamesByRound[0].forEach(game => {
          game.sides.home = {
            sourceGame: newPoolBrackets[poolIndex].game,
            score: { score: null },
            pool: newPoolBrackets[poolIndex].pool
          };
          poolIndex++;
          if (game.sides.away) {
            game.sides.away = {
              sourceGame: newPoolBrackets[poolIndex].game,
              score: { score: null },
              pool: newPoolBrackets[poolIndex].pool
            };
            poolIndex++;
          }
        });

        newBrackets[poolFinalIndex] = {
          ...newBrackets[poolFinalIndex],
          game: playoffStruct.finalGame,
          gamesByRound: playoffStruct.gamesByRound,
          outcomes: {}
        };
      }
    } else {
      // === SINGLE BRACKET SHUFFLE ===
      const shuffledPlayers = [...bracket.shuffledPlayers].sort(() => Math.random() - 0.5);
      const { finalGame, gamesByRound } = generateSingleEliminationGameStructure(shuffledPlayers);

      newBrackets[bracketIndex] = {
        ...bracket,
        shuffledPlayers,
        game: finalGame,
        gamesByRound,
        outcomes: {}
      };
    }

    return newBrackets;
  });

  // === OUTCOMES CLEAR FOR SHUFFLED BRACKET(S) ===
  setBracketsOutcomes(prev => {
    const newOutcomes = { ...prev };

    const bracket = brackets.find(b => b.key === bracketKey);

    if (bracket?.pool) {
      // Pool category → सारे related brackets के outcomes clear
      const baseKey = bracketKey.replace(/_Pool.*$/, '');
      Object.keys(newOutcomes).forEach(key => {
        if (key.startsWith(baseKey)) {
          newOutcomes[key] = {};
        }
      });
    } else {
      // Single bracket
      newOutcomes[bracketKey] = {};
    }

    return newOutcomes;
  });

class ErrorBoundary extends React.Component {
  state = { hasError: false, error: null };
  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }
  render() {
    if (this.state.hasError) {
      return (
        <div className={styles.errorBoundary}>
          Something went wrong: {this.state.error?.message || 'Unknown error'}. Please refresh.
        </div>
      );
    }
    return this.props.children;
  }
}

          ...(tournamentData.weightCategories?.selected?.male || []),
          ...(tournamentData.weightCategories?.selected?.female || [])
        ]);


        const mappedPlayers = playersData.map((p) => ({
          ...p,
          gender: p.gender ? p.gender.charAt(0).toUpperCase() + p.gender.slice(1).toLowerCase() : '',
          ageCategory: ageCategoryMapping[normalizeString(p.ageCategory)] || p.ageCategory,
          weightCategory: p.weightCategory,
          name: p.name || 'TBD',
          team: p.team || '',
        })).filter((p) => {
          const genderOk = p.gender && allowedGenders.has(p.gender);
          const ageOk = p.ageCategory && allowedAgeCategories.has(p.ageCategory);
          const weightOk = [...allowedWeightCategories].some(
            (wc) => normalizeString(wc) === normalizeString(p.weightCategory)
          );
          return genderOk && ageOk && weightOk;
        });

        const genders = [...allowedGenders].sort((a, b) => genderOrder.indexOf(a) - genderOrder.indexOf(b));
        const ageCategories = [...new Set([...allowedAgeCategories].map((ac) => ageCategoryMapping[normalizeString(ac)] || ac))].sort(
          (a, b) => ageCategoryOrder.indexOf(a) - ageCategoryOrder.indexOf(b)
        );

        setAvailableGenders(genders.length > 0 ? genders : ['Male', 'Female']);
        setSelectedGenders(genders.length > 0 ? genders : ['Male']);
        setAvailableAgeCategories(ageCategories.length > 0 ? ageCategories : ageCategoryOrder);
        setSelectedAgeCategories(ageCategories.length > 0 ? ageCategories : ageCategoryOrder);
        setPlayers(mappedPlayers);

     if (initialLoadRef.current) {
        initialLoadRef.current = false;
        console.log('Attempting to restore saved tie-sheet from server...');

        try {
          const tieSheetResponse = await axios.get(`${baseUrl}/tournament/${id}/tiesheet`, config);
          const saved = tieSheetResponse.data?.tiesheet;

          if (saved?.brackets?.length > 0) {
            console.log(`Restored ${saved.brackets.length} brackets from server`);
            dispatch(
              setInitialBrackets({
                brackets: saved.brackets,
                outcomes: saved.outcomes || {},
              })
            );

            // Restore filters if they exist
            if (saved.filters) {
              setSelectedGenders((prev) => saved.filters.selectedGenders || prev);
              setSelectedAgeCategories((prev) => saved.filters.selectedAgeCategories || prev);
            }

            console.log('TieSheet successfully restored from server');
          } else {
            console.log('No saved brackets found on server (normal on first use)');
          }
        } catch (err) {
          if (err.response?.status === 404) {
            console.log('No saved tie-sheet exists on server yet (404 - expected)');
          } else {
            console.warn('Failed to load saved tie-sheet:', err.message);
          }
          // → Do NOT set error here — we can still generate brackets locally
        }
      }

      setIsLoading(false);
    } catch (err) {
      console.error('Main data fetch failed:', err);
      if (isMounted) {
        setError(err.message || 'Failed to load tournament or player data');
        setIsLoading(false);
      }
    }
  };

  fetchData();

  return () => {
    isMounted = false;
  };
}, [id, token, location.state?.players, dispatch]);
  // Real-time sync from entry page
  useEffect(() => {
    const handleEntryUpdate = (e) => {
      console.log('EVENT RECEIVED: entryDataUpdated_', id);
      const saved = localStorage.getItem(`entryData_${id}`);
      if (saved) {
        try {
          const updatedPlayers = JSON.parse(saved);
          const filteredPlayers = updatedPlayers.filter(
            (p) => p?.name && p?.gender && p?.ageCategory && p?.weightCategory
          );
          setPlayers(filteredPlayers);
          // Optional: if you want to regenerate brackets on entry change
          // dispatch(setBrackets([]));
          // dispatch(setOutcomes({}));
        } catch (err) {
          console.error('Failed to parse updated entry data', err);
        }
      }
    };

    window.addEventListener(`entryDataUpdated_${id}`, handleEntryUpdate);
    return () => {
      window.removeEventListener(`entryDataUpdated_${id}`, handleEntryUpdate);
    };
  }, [id, dispatch]);

  
  // Filtered brackets from Redux
  const filteredBrackets = useMemo(
  () =>
    brackets.filter(
      (b) =>
        selectedGenders.includes(b.gender) &&
        selectedAgeCategories.includes(b.ageCategory)
    ),
  [brackets, selectedGenders, selectedAgeCategories, shuffleVersion] // ← add this
);

  // Debounced auto-save (structure only)
  const debouncedSaveTieSheet = useMemo(
    () =>
      debounce(async () => {
        if (!brackets.length || !id || isLoading) return;
        try {
          const baseUrl = import.meta.env.VITE_API_URL || 'http://localhost:5000/api';
          const config = {
            headers: {
              Authorization: `Bearer ${token}`,
              'Content-Type': 'application/json',
            },
            timeout: 30000,
          };
          const dataToSave = {
            tiesheet: {
              brackets,
              filters: { selectedGenders, selectedAgeCategories },
            },
          };
          await axios.put(`${baseUrl}/tournament/${id}/tiesheet`, dataToSave, config);
          console.log('TieSheet auto-saved');
        } catch (err) {
          console.error('Auto-save failed:', err);
        }
      }, 8000),
    [brackets, selectedGenders, selectedAgeCategories, id, token, isLoading]
  );

  useEffect(() => {
    debouncedSaveTieSheet();
    return () => debouncedSaveTieSheet.cancel();
  }, [debouncedSaveTieSheet]);

  // Sync outcomes to localStorage (optional – can remove if Redux persist is used)
  useEffect(() => {
    localStorage.setItem(`brackets_outcomes_${id}`, JSON.stringify(bracketsOutcomes));
  }, [bracketsOutcomes, id]);

// Outcomes ko localStorage se load aur validate karne wala (sirf ek useEffect)
useEffect(() => {
  const key = `brackets_outcomes_${id}`;
  const saved = localStorage.getItem(key);

  if (saved) {
    try {
      const parsed = JSON.parse(saved);

      // Basic check – agar parsed object nahi to error
      if (typeof parsed !== 'object' || parsed === null) {
        throw new Error('Parsed outcomes is not a valid object');
      }

      // Existing brackets ko preserve karte hue sirf outcomes update karo
      dispatch(setInitialBrackets({
        brackets: brackets,   // current brackets same rakhna
        outcomes: parsed      // naye outcomes load karo
      }));

      console.log(`Loaded valid outcomes for tournament ${id}`);
    } catch (err) {
      console.warn(`Failed to parse ${key} – clearing invalid data`, err.message);
      
      // Corrupted data hata do taaki baar baar na dikhe
      localStorage.removeItem(key);
      
      // Safe fallback – empty outcomes set kar do
      dispatch(setInitialBrackets({
        brackets: brackets,
        outcomes: {}
      }));
    }
  }
}, [id, dispatch, brackets]); // brackets dependency add ki taaki stale value na ho

  // Refresh function
 const handleRefresh = async () => {
  try {
    setIsLoading(true);
    setError(null);
dispatch(setInitialBrackets({ brackets: [], outcomes: {} }));
    // Step 1: Latest players reload karo entry se (localStorage)
const savedData = localStorage.getItem(`entryData_${id}`);
    let playersData = [];
    if (savedData) {
      playersData = JSON.parse(savedData);
    }

    // Filter valid players
    const mappedPlayers = playersData
      .map((p) => ({
        ...p,
        gender: p.gender ? p.gender.charAt(0).toUpperCase() + p.gender.slice(1).toLowerCase() : '',
        ageCategory: ageCategoryMapping[normalizeString(p.ageCategory)] || p.ageCategory,
        weightCategory: p.weightCategory,
        name: DOMPurify.sanitize(p.name || 'TBD'),
        team: DOMPurify.sanitize(p.team || ''),
      }))
      .filter((p) => p?.name && p?.gender && p?.ageCategory && p?.weightCategory);

    setPlayers(mappedPlayers);

    // Step 2: Sirf LOCKED brackets + unke outcomes preserve karo
    const preservedBrackets = brackets.filter((b) => lockedBrackets.has(b.key));
    const preservedOutcomes = {};
    preservedBrackets.forEach((b) => {
      if (bracketsOutcomes[b.key]) {
        preservedOutcomes[b.key] = { ...bracketsOutcomes[b.key] };
      }
    });

    // Step 3: Server pe CLEAR mat karo — preserved state bhejo
    const baseUrl = import.meta.env.VITE_API_URL || 'http://localhost:5000/api';
    const config = { headers: { Authorization: `Bearer ${token}` } };

    // await axios.put(
    //   `${baseUrl}/tournament/${id}/tiesheet`,
    //   {
    //     tiesheet: {
    //       brackets: preservedBrackets,          // ← yeh rakho (empty mat bhejo)
    //       outcomes: preservedOutcomes,
    //       filters: { selectedGenders, selectedAgeCategories },
    //     },
    //   },
    //   config
    // );

    console.log(`Refresh: Preserved ${preservedBrackets.length} locked brackets`);

    // Step 4: Redux mein preserved state set kar do
    // dispatch(setInitialBrackets({
    //   brackets: preservedBrackets,
    //   outcomes: preservedOutcomes,
    // }));

  //  setIsLoading(false);
  } catch (err) {
    console.error('Refresh failed:', err);
    setError('Refresh failed. Try again.');
    setIsLoading(false);
  }
};

  // Print All Brackets (unchanged)
  const printAllBrackets = () => {
    if (!filteredBrackets.length) {
      alert('No brackets to print');
      return;
    }

    setIsLoading(true);

    try {
      const allPages = filteredBrackets
        .map((b) => document.querySelector(`#bracket-${sanitizeId(b.key)}`)?.closest('.page'))
        .filter(Boolean);

      if (allPages.length === 0) {
        alert('No printable brackets found');
        setIsLoading(false);
        return;
      }

      document.body.classList.add('print-all-mode');

      setTimeout(() => {
        window.print();

        setTimeout(() => {
          document.body.classList.remove('print-all-mode');
          setIsLoading(false);
        }, 1000);
      }, 600);

      if (typeof window.saveTieSheetRecord === 'function') {
        const allHtml = allPages.map((p) => p.outerHTML).join('<div style="page-break-after: always;"></div>');
        window.saveTieSheetRecord('ALL_BRACKETS', allHtml, {
          gender: 'All',
          ageCategory: 'All',
          weightCategory: 'All',
          pool: '',
          categoryPlayerCount: players.length,
        });
      }
    } catch (err) {
      console.error('Print All failed:', err);
      alert('Failed to print all. Try single bracket.');
      setIsLoading(false);
    }
  };

  // Save All to PDF (unchanged)
  const saveAllToPDF = async () => {
    if (!filteredBrackets.length) {
      alert('No brackets to save');
      return;
    }

    try {
      const doc = await createPDFDoc(filteredBrackets);
      doc.save(`all_brackets_${new Date().toISOString().slice(0, 10)}.pdf`);

      if (typeof window.saveTieSheetRecord === 'function') {
        const allHtml = filteredBrackets
          .map((b) => {
            const el = document.querySelector(`#bracket-${sanitizeId(b.key)}`);
            return el ? el.outerHTML : '';
          })
          .join('<div style="page-break-after: always;"></div>');

        window.saveTieSheetRecord('ALL_BRACKETS', allHtml, {
          gender: 'All',
          ageCategory: 'All',
          weightCategory: 'All',
          pool: '',
          categoryPlayerCount: players.length,
        });
      }
    } catch (err) {
      console.error('Save All PDF failed:', err);
      alert('PDF generation failed. Use Print All instead.');
    }
  };

  if (!isAuthenticated) {
    navigate('/login');
    return null;
  }

  return (
    <ErrorBoundary>
      <div className={styles.tieSheetContainer}>
        {isLoading && (
          <div className={styles.loadingOverlay}>
            <div className={styles.spinner}></div>
            <p>Loading Tie Sheet...</p>
          </div>
        )}

        {error && (
          <div className={styles.errorMessage}>
            <h2>Error</h2>
            <p>{error}</p>
            <button className={styles.toggleButton} onClick={() => navigate(`/tournaments/${id}/entry`)}>
              Go to Entry Page
            </button>
          </div>
        )}

        {!isLoading && !error && (
          <>
            <div className={styles.stickyWrapper}>
              <div className={styles.pdfbtn}>
                <h2 className={`${styles.text2xl} ${styles.fontBold} ${styles.textCenter}`}>
                  Tie Sheet for {tournamentName}
                </h2>

                <div className={styles.actionButtons}>
                  <button className={styles.toggleButton} onClick={handleRefresh} disabled={isLoading}>
                    Refresh
                  </button>
                  <button
                    className={styles.toggleButton}
                    onClick={printAllBrackets}
                    disabled={!filteredBrackets.length || isLoading}
                  >
                    Print All
                  </button>
                  <button
                    className={styles.toggleButton}
                    onClick={saveAllToPDF}
                    disabled={!filteredBrackets.length || isLoading}
                  >
                    Save All to PDF
                  </button>
                </div>
              </div>

              <BracketFilters
                availableGenders={availableGenders}
                selectedGenders={selectedGenders}
                setSelectedGenders={setSelectedGenders}
                availableAgeCategories={availableAgeCategories}
                selectedAgeCategories={selectedAgeCategories}
                setSelectedAgeCategories={setSelectedAgeCategories}
              />
            </div>

            {filteredBrackets.length === 0 ? (
              <p className={styles.noData}>No brackets available for the selected categories.</p>
            ) : (
              filteredBrackets.map((bracket, index) => {
                const baseKey = bracket.key.replace(/_Pool.*$/, '');
                // Medals can still be computed locally or move to selector
                const medals = {
                  gold: MEDAL_PLACEHOLDER,
                  silver: MEDAL_PLACEHOLDER,
                  bronze1: MEDAL_PLACEHOLDER,
                  bronze2: MEDAL_PLACEHOLDER,
                };

                const bracketSizeClass =
                  bracket.playerCount === 1
                    ? styles.singlePlayer
                    : bracket.playerCount === 2
                    ? styles.twoPlayer
                    : bracket.playerCount === 3
                    ? styles.threePlayer
                    : bracket.playerCount <= 8
                    ? styles.smallBracket
                    : styles.multiPlayer;

                return (
                  <div key={bracket.key} className={`${styles.bracketWrapper} ${bracketSizeClass}`}>
                    <BracketActions
                      bracket={bracket}
                      lockedBrackets={lockedBrackets}
                      brackets={brackets}
                      bracketsOutcomes={bracketsOutcomes}
                      filteredBrackets={filteredBrackets}
                      tournamentId={id}
                      toggleLock={handleToggleLock}
                      // No skipNextGeneration anymore
                    />

                    <div className={`${styles.page} ${bracketSizeClass}`} id={`bracket-${sanitizeId(bracket.key)}`}>
                      <BracketHeader
                        bracket={bracket}
                        tournamentName={tournamentName}
                        federation={federation}
                        logoLeft={logoLeft}
                        logoRight={logoRight}
                      />

                      {bracket.gamesByRound?.length > 0 ? (
                        <BracketTable
                          bracket={bracket}
                          bracketsOutcomes={bracketsOutcomes}
                          className={
                            bracket.playerCount === 1
                              ? styles.singlePlayer
                              : bracket.playerCount === 2
                              ? styles.twoPlayer
                              : bracket.playerCount === 3
                              ? styles.threePlayer
                              : ''
                          }
                          lockedBrackets={lockedBrackets}
                        />
                      ) : (
                        <p>No players available for this category.</p>
                      )}

                      <div className={styles.signatureMedalSection}>
                        <MedalSection
                          medals={medals}
                          categoryPlayerCount={bracket.categoryPlayerCount || bracket.playerCount}
                          bracket={bracket}
                          bracketsOutcomes={bracketsOutcomes}
                        />
                        <SignatureSection />
                      </div>

                      <BracketFooter index={index} total={filteredBrackets.length} />
                    </div>
                  </div>
                );
              })
            )}
          </>
        )}
      </div>
    </ErrorBoundary>
  );
};
